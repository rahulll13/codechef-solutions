import sys
def solve():
    R1, R2, R3 = map(int, input().split())
    
    if R1 > R2 + R3:
        print("YES")
    elif R2 > R1 + R3:
        print("YES")
    elif R3 > R1 + R2:
        print("YES")
    else:
        print("NO")
        
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
