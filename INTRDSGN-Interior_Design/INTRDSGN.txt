import sys
def solve():
    X1,Y1,X2,Y2 = map(int, input().split())
    
    if (X1 + Y1) < (X2 + Y2):
        print(X1 + Y1)
    else:
        print(X2 + Y2)
        
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except (EOFError, ValueError):
        pass
