import sys
def solve():
    X,Y = map(int, input().split())
    
    if X > Y:
        print(X - Y)
    else:
        print(Y - X)
    
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except (EOFError, ValueError):
        pass
