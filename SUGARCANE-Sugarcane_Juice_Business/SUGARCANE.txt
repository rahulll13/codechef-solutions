import sys
def solve():
    N = int(input())
    
    Tc= N * 50
    bs = int(0.2 * Tc)
    bs_ml = int(0.2 * Tc)
    sr = int(0.3 * Tc)
    Tp = Tc - (bs + bs_ml + sr)
    
    print(Tp)
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
