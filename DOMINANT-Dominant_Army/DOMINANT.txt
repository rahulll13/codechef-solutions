import sys
def solve():
    A,B,C = map(int, input().split())
    
    if A > B + C:
        print("YES")
    elif B > A + C:
        print("YES")
    elif C > A + B:
        print("YES")
    else:
        print("NO")
        
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
