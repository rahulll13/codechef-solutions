import sys
def solve():
    X = int(input())
    
    amt = 0
    if X <= 100:
        amt = X 
    elif X > 100 and X <= 1000:
        amt = X - 25
    elif X > 1000 and X <= 5000:
        amt = X - 100 
    elif X > 5000:
        amt = X - 500
    
    print(amt)
    
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
