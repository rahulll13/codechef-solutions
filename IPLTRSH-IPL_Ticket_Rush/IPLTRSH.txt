import sys
def solve():
    N, M = map(int, input().split())
    if M >= N:
        print("0")
    else:
        print(N - M)
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except (EOFError, ValueError):
        pass
