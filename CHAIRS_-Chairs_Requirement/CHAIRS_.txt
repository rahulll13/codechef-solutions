import sys
def solve():
    X, Y = map(int, input().split())
    
    if X >= Y:
        print(X - Y)
    
    if X < Y:
        print("0")
    
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except (EOFError, ValueError):
        pass
