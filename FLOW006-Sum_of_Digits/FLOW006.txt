import sys
def solve():
    N = int(input())
    num = N
    summ = 0
    
    while num > 0:
        digits = num % 10
        summ = summ + digits
        num = num // 10
        
    print(summ)
        
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
