import sys
def solve():
    X,Y,Z = map(int, input().split())
    
    total_seat = 10 * X 
    ticket_available = min(Y, total_seat)
    ans = ticket_available * Z 
    print(ans)
    
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
