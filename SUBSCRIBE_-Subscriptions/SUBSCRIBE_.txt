import sys, math
def solve():
    N,X = map(int, input().split())
    
    subscription = math.ceil(N/6)
    Total_cost = subscription * X   
    
    print(Total_cost)

if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
