import sys
def solve():
    A,B,C,D = map(int, input().split())
    Dis_1 = A - C 
    Dis_2 = B - D 
    
    if Dis_1 < Dis_2:
        print("First")
    elif Dis_1 > Dis_2:
        print("second")
    else:
        print("Any")
        
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except (EOFError, ValueError):
        pass
