import sys
def solve():
    X,Y,Z = map(int, input().split())
    
    if X + Y <= Z:
        print("2")
    elif  X <= Z:
        print("1")
    else:
        print("0")
        
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
