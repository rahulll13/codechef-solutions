import sys
def solve():
    X, Y = map(int, input().split())
    
    if X < Y:
        print("PROFIT")
    elif X == Y:
        print("NEUTRAL")
    else:
        print("LOSS")

if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except (EOFError, ValueError):
        pass
