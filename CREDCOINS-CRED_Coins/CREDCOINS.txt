import sys, math
def solve():
    X,Y = map(int, input().split())
    
    coins_recived = X * Y
    
    print(math.floor(coins_recived/100))
    
if __name__ == "__main__":
    try:
        T = int(input())
        for _ in range(T):
            solve()
    except(EOFError, ValueError):
        pass
